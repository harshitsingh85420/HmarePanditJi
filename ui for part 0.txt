    /**
     * Generates scattered confetti shapes around the center icon
     * Shapes: Circles and Rectangles
     * Colors: Saffron (#F09942), Gold (#FFD700), White (#FFFFFF), Success Green (#10B981)
     */
    function createConfetti() {
      const container = document.getElementById('confetti-container');
      const colors = ['#F09942', '#FFD700', '#FFFFFF', '#10B981'];
      const shapes = ['rounded-full', '']; // circle or rectangle
      
      for (let i = 0; i < 20; i++) {
        const confetti = document.createElement('div');
        const size = Math.random() * 6 + 4;
        const color = colors[Math.floor(Math.random() * colors.length)];
        const shape = shapes[Math.floor(Math.random() * shapes.length)];
        
        confetti.className = `confetti absolute ${shape}`;
        confetti.style.width = `${size}px`;
        confetti.style.height = `${size}px`;
        confetti.style.backgroundColor = color;
        
        // Positioning in a burst pattern
        const angle = Math.random() * Math.PI * 2;
        const radius = Math.random() * 80 + 40;
        const x = Math.cos(angle) * radius;
        const y = Math.sin(angle) * radius;
        
        confetti.style.left = `calc(50% + ${x}px)`;
        confetti.style.top = `calc(50% + ${y}px)`;
        
        // Randomize animation duration and delay
        confetti.style.animationDuration = (Math.random() * 2 + 2) + 's';
        confetti.style.animationDelay = (Math.random() * 1.5) + 's';
        
        container.appendChild(confetti);
      }
    }

    // Initialize confetti on load
    window.addEventListener('load', createConfetti);
  </script>

<script data-purpose="confetti-generator">
    /**
     * Generates scattered confetti shapes around the center icon
     * Shapes: Circles and Rectangles
     * Colors: Saffron (#F09942), Gold (#FFD700), White (#FFFFFF), Success Green (#10B981)
     */
    function createConfetti() {
      const container = document.getElementById('confetti-container');
      const colors = ['#F09942', '#FFD700', '#FFFFFF', '#10B981'];
      const shapes = ['rounded-full', '']; // circle or rectangle
      
      for (let i = 0; i < 20; i++) {
        const confetti = document.createElement('div');
        const size = Math.random() * 6 + 4;
        const color = colors[Math.floor(Math.random() * colors.length)];
        const shape = shapes[Math.floor(Math.random() * shapes.length)];
        
        confetti.className = `confetti absolute ${shape}`;
        confetti.style.width = `${size}px`;
        confetti.style.height = `${size}px`;
        confetti.style.backgroundColor = color;
        
        // Positioning in a burst pattern
        const angle = Math.random() * Math.PI * 2;
        const radius = Math.random() * 80 + 40;
        const x = Math.cos(angle) * radius;
        const y = Math.sin(angle) * radius;
        
        confetti.style.left = `calc(50% + ${x}px)`;
        confetti.style.top = `calc(50% + ${y}px)`;
        
        // Randomize animation duration and delay
        confetti.style.animationDuration = (Math.random() * 2 + 2) + 's';
        confetti.style.animationDelay = (Math.random() * 1.5) + 's';
        
        container.appendChild(confetti);
      }
    }

    // Initialize confetti on load
    window.addEventListener('load', createConfetti);
  </script>
